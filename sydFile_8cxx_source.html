65 void syd::File::OnDelete(syd::Database * d)
66 {
67  syd::StandardDatabase * db = dynamic_cast<syd::StandardDatabase*>(d);
68 
69  // When a File is deleted, we also delete the file on disk
70  std::string f = db->GetAbsolutePath(*this);
71  db->AddFileToDelete(f);// list_of_files_to_delete.push_back(f);
72 
73  // We also look for images to be deleted. Manual search because dont know how to do
74  std::vector<syd::Image> images_temp;
75  std::vector<syd::Image> images;
76  db->Query<syd::Image>(images_temp);
77  for(auto i:images_temp) {
78  for(auto f:i.files) {
79  if (f->id == id) images.push_back(i);
80  }
81  }
82  for(auto i:images) db->AddToDeleteList(i);
83 
84 
85  // We also look for DicomFiles to be deleted.
86  std::vector<syd::DicomFile> dicomfiles;
87  db->Query<syd::DicomFile>(odb::query<syd::DicomFile>::file == id, dicomfiles);
88  for(auto i:dicomfiles) db->AddToDeleteList(i);
89 }
