36 int main(int argc, char* argv[])
37 {
38  // Init
39  SYD_INIT_GGO(syd_transformix, 0);
40 
41  // Load plugin
42  syd::PluginManager::GetInstance()->Load();
43  syd::DatabaseManager* m = syd::DatabaseManager::GetInstance();
44 
45  // Get the database
46  syd::StandardDatabase * db = m->Read<syd::StandardDatabase>(args_info.db_arg);
47 
48  // Read the ImageTransform
49  std::vector<syd::IdType> tids;
50  std::vector<std::string> stids;
51  syd::GetWords(stids, args_info.transform_arg);
52  for(auto s:stids) tids.push_back(atoi(s.c_str()));
53  syd::ImageTransform::vector initial_transforms;
54  db->Query(initial_transforms, tids);
55 
56  // Read ids from the command line and the pipe
57  std::vector<syd::IdType> ids;
58  syd::ReadIdsFromInputPipe(ids); // Read the standard input if pipe
59  for(auto i=0; i<args_info.inputs_num; i++)
60  ids.push_back(atoi(args_info.inputs[i]));
61  if (ids.size() < 1) {
62  LOG(FATAL) << "Please provide at least one image id.";
63  }
64  syd::Image::vector initial_images;
65  db->Query(initial_images, ids);
66 
67  // Match
68  syd::Image::vector images;
69  syd::ImageTransform::vector transforms;
70  for(auto i=0; i<initial_images.size(); i++) {
71  syd::Image::pointer image = initial_images[i];
72  // Find the corresponding transforms
73  int found = -1;
74  for(auto j=0; j<initial_transforms.size(); j++) {
75  if (initial_transforms[j]->moving_image->frame_of_reference_uid == image->frame_of_reference_uid) {
76  if (found != -1) {
77  LOG(FATAL) << "Error two transform could be used with the image: " << image << std::endl
78  << "First is: " << initial_transforms[found] << std::endl
79  << "Second is: " << initial_transforms[j] << std::endl;
80  }
81  else found = j;
82  }
83  }
84  if (found == -1) {
85  LOG(FATAL) << "Error, could not find adequate ImageTransform for image:" << image;
86  }
87  images.push_back(image);
88  transforms.push_back(initial_transforms[found]);
89  }
90 
91  // Loop
92  for(auto i=0; i<images.size(); i++) {
93 
94  // Get the input and the transform
95  syd::Image::pointer input_image = images[i];
96  syd::ImageTransform::pointer transform = transforms[i];
97 
98  std::string transform_path = db->GetAbsolutePath(transform->transform_file);
99  std::string input_image_path = db->GetAbsolutePath(input_image);
100 
101  // Modify the transform file according to the input image spacing
102  std::ifstream in(transform_path);
103  transform_path = transform_path+"_temp.txt";
104  std::ofstream out(transform_path);
105  std::string line;
106  while (std::getline(in, line)) {
107  syd::Replace(line, "(Size ", "//(Size ");
108  syd::Replace(line, "(Spacing ", "//(Spacing ");
109  syd::Replace(line, "(FixedInternalImagePixelType ", "//(FixedInternalImagePixelType ");
110  syd::Replace(line, "(MovingInternalImagePixelType ", "//(MovingInternalImagePixelType ");
111  syd::Replace(line, "(DefaultPixelValue ", "//(DefaultPixelValue ");
112  syd::Replace(line, "(ResultImagePixelType ", "//(ResultImagePixelType ");
113  out << line << std::endl;
114  }
115  out << "(Size "
116  << input_image->size[0] << " "
117  << input_image->size[1] << " "
118  << input_image->size[2] << ")" << std::endl;
119  out << "(Spacing "
120  << input_image->spacing[0] << " "
121  << input_image->spacing[1] << " "
122  << input_image->spacing[2] << ")" << std::endl;
123  // out << "(FixedInternalImagePixelType " << input_image->pixel_type << ")" << std::endl;
124  // out << "(MovingInternalImagePixelType " << input_image->pixel_type << ")" << std::endl;
125  if (input_image->pixel_type == "short") out << "(DefaultPixelValue -1000)" << std::endl; // FIXME
126  else out << "(DefaultPixelValue 0)" << std::endl;
127  out << "(ResultImagePixelType " << input_image->pixel_type << ")" << std::endl;
128  out.close();
129 
130 
131  // Check frame_of_reference_uid
132  if (input_image->frame_of_reference_uid != transform->moving_image->frame_of_reference_uid) {
133  LOG(WARNING) << "Error the frame_of_reference_uid of the input image is different from the one of the moving_image of the transform" << std::endl
134  << "input : " << input_image << std::endl
135  << "moving_image : " << transform->moving_image << std::endl;
136  continue;
137  }
138 
139  // Create output image
140  syd::Image::pointer output_image;
141  db->New(output_image);
142  output_image->CopyFrom(input_image);
143  output_image->tags.clear(); // dont herit tag
144  output_image->frame_of_reference_uid = transform->fixed_image->frame_of_reference_uid; // copy the ref from the fixed_image
145  db->Insert(output_image);
146  std::ostringstream oss;
147  if (input_image->dicoms.size() == 0) oss << "IMAGE_" << output_image->id << ".mhd";
148  else oss << input_image->dicoms[0]->dicom_modality << "_" << output_image->id << ".mhd";
149  std::string mhd_filename = oss.str();
150  std::string mhd_relative_path = output_image->ComputeRelativeFolder()+PATH_SEPARATOR;
151  std::string mhd_path = db->ConvertToAbsolutePath(mhd_relative_path+mhd_filename);
152  output_image->UpdateFile(db, mhd_filename, mhd_relative_path);
153  std::string output_image_path = db->GetAbsolutePath(output_image);
154 
155  // Tag ? ct spect mask to copy ; transform to add
156  syd::Tag::vector tags;
157  db->FindTags(tags, args_info.tags_arg);
158  for(auto t:tags) output_image->AddTag(t);
159 
160  db->Update(output_image);
161 
162  // Create command line
163  std::ostringstream cmd;
164  cmd << "transformix -in " << input_image_path
165  << " -tp " << transform_path
166  << " -out " << db->ConvertToAbsolutePath(output_image->ComputeRelativeFolder());
167  // cmd << options; // additional options to transformix
168 
169  // Execute transformix
170  LOG(1) << cmd.str();
171  int r = syd::ExecuteCommandLine(cmd.str(), args_info.verbose_arg);
172 
173  // Get result path
174  std::string f = db->ConvertToAbsolutePath(output_image->ComputeRelativeFolder()+PATH_SEPARATOR+"result.mhd");
175 
176  if (r!=0 || !fs::exists(f)) { // fail
177  LOG(1) << "Command fail, removing temporary image";
178  db->Delete(output_image);
179  }
180  else {
181  syd::RenameMHDImage(f, output_image_path);
182  LOG(1) << "Image computed. Result: " << output_image;
183  }
184  }
185 
186  // This is the end, my friend.
187 }
