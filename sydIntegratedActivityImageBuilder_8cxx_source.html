154 void syd::IntegratedActivityImageBuilder::CreateIntegratedActivityImage()
155 {
156  // typedef
157  typedef float PixelType;
158  typedef itk::Image<PixelType,3> ImageType;
159 
160  // Initialisation
161  tac_image_ = Image4DType::New();
162  InitInputData();
163 
164  // create initial tac with the times
165  TimeActivityCurve tac;
166  for(auto t:times_) tac.AddValue(t, 0.0);
167 
168  // Init solver
169  InitSolver();
170 
171  // Init main iterator
172  typedef itk::ImageRegionIterator<Image4DType> Iterator4D;
173  Iterator4D it(tac_image_, tac_image_->GetLargestPossibleRegion());
174 
175  // Init mask iterator
176  bool mask_flag = false;
177  Iterator3D it_mask;
178  if (mask_) {
179  mask_flag = true;
180  it_mask = Iterator3D(mask_, mask_->GetLargestPossibleRegion());
181  it_mask.GoToBegin();
182  }
183 
184  // Init output iterators
185  for(auto & o:outputs_) {
186  o->iterator.GoToBegin();
187  o->use_current_tac = restricted_tac_flag_;
188  }
189 
190  // debug init, sort point by index
191  bool debug_this_point_flag;
192  int debug_point_current=0;
193  std::sort(begin(debug_data), end(debug_data),
194  [](DebugType a, DebugType b) {
195  return a.index < b.index; });
196  if(debug_data.size() == 0) debug_point_current = -1;
197 
198  // log
199  // models, threshold, mask
200  std::string sm;
201  for(auto m:models_) sm += m->name_+" ";
202  LOG(1) << "Starting fit with models : " << sm << "; "
203  << (mask_flag ? "mask":"no_mask")
204  << " R2_min = " << R2_min_threshold_
205  << (restricted_tac_flag_ ? " with last 3 points of the curve only":"");
206 
207  // main loop
208  int x = 0;
209  int n = images_[0]->GetLargestPossibleRegion().GetNumberOfPixels();
210  int number_of_pixels_in_mask = 0;
211  int number_of_pixels_success = 0;
212  for (it.GoToBegin(); !it.IsAtEnd(); ) {
213 
214  // Consider current point, is it a debug point ?
215  if (debug_point_current != -1 and x == debug_data[debug_point_current].index) debug_this_point_flag = true;
216  else debug_this_point_flag = false;
217 
218  // Check if pixel is in the mask
219  if (((mask_flag and it_mask.Get() == 0) and !debug_this_point_flag)
220  or
221  (debug_only_flag_ and !debug_this_point_flag)) { // skip it
222  for(auto i=0; i<images_.size(); i++) ++it;
223  }
224  else {
225  ++number_of_pixels_in_mask;
226  // Create current tac
227  for(auto i=0; i<images_.size(); i++) {
228  tac.SetValue(i, it.Get());
229  ++it; // next value
230  }
231 
232  syd::TimeActivityCurve restricted_tac;
233  if (restricted_tac_flag_) {
234  // Select only the end of the curve (min 2 points);
235  auto m = tac.FindMaxIndex();
236  m = std::min(m, tac.size()-2);
237  for(auto i=m; i<tac.size(); i++)
238  restricted_tac.AddValue(tac.GetTime(i), tac.GetValue(i));
239  }
240 
241  // Solve
242  int best;
243  if (restricted_tac_flag_)
244  best = FitModels(restricted_tac, debug_this_point_flag, &debug_data[debug_point_current]);
245  else
246  best = FitModels(tac, debug_this_point_flag, &debug_data[debug_point_current]);
247 
248  // Set the current selected model, update the output
249  if (best != -1) {
250  ++number_of_pixels_success;
251  current_model_ = models_[best];
252  // Update output
253  for(auto o:outputs_) o->Update(tac, current_model_);
254  }
255 
256  // debug points for plot
257  if (debug_this_point_flag) {
258  debug_data[debug_point_current].selected_model = best;
259  debug_data[debug_point_current].tac = tac;
260  ++debug_point_current;
261  }
262  }
263 
264  // Next debug images
265  for(auto o:outputs_) ++o->iterator;
266 
267  // progress bar
268  ++x;
269  loadbar(x,n);
270 
271  // Next in mask
272  if (mask_flag) ++it_mask;
273  }
274 
275  LOG(2) << "Done. " << x << " pixels " << number_of_pixels_in_mask << " pixels in mask "
276  << number_of_pixels_success << " successfully fit "
277  << number_of_pixels_in_mask-number_of_pixels_success << " fail";
278 }

283 int syd::IntegratedActivityImageBuilder::FitModels(TimeActivityCurve & tac,
284  bool debug_this_point_flag,
285  DebugType * debug_current)
286 {
287  for(auto model:models_) {
288  ceres::Problem problem;// New problem each time. (I did not manage to change that)
289  model->SetProblemResidual(&problem, tac);
290  ceres::Solve(*ceres_options_, &problem, &model->ceres_summary_); // Go !
291 
292  if (debug_this_point_flag) {
293  syd::FitModelBase * mm = model->Clone();
294  debug_current->models.push_back(mm);
295  // Copy the current tac
296  syd::TimeActivityCurve * c = new TimeActivityCurve;
297  c->CopyFrom(tac);
298  mm->current_tac = c;
299  }
300  }
301 
302  // Select the best model
303  int best = -1;
304  double R2_threshold = R2_min_threshold_;
305  double best_AICc = 666;
306  double best_R2 = 0.0;
307  for(auto i=0; i<models_.size(); i++) {
308  auto & m = models_[i];
309  if (m->IsAcceptable()) {
310  double R2 = m->ComputeR2(tac);
311  if (R2 > R2_threshold and R2 > best_R2) {
312  double AICc;
313  bool b = m->IsAICcValid(tac.size());
314  if (b) AICc = m->ComputeAICc(tac);
315  if (!b or AICc < best_AICc) { // if AICc not valid, consider it is ok
316  best = i;
317  best_AICc = AICc;
318  best_R2 = R2;
319  }
320  }
321  }
322  }
323  return best;
324 }

329 void syd::IntegratedActivityImageBuilder::InitInputData()
330 {
331  // consider images_ OR create a 4D images ?
332  typename Image4DType::SizeType size;
333  for(auto i=1; i<4; i++) size[i] = images_[0]->GetLargestPossibleRegion().GetSize()[i-1];
334  size[0] = images_.size();
335  typename Image4DType::RegionType region;
336  region.SetSize(size);
337  tac_image_->SetRegions(region);
338  typename Image4DType::SpacingType spacing;
339  for(auto i=1; i<4; i++) spacing[i] = images_[0]->GetSpacing()[i-1];
340  spacing[0] = 1.0;
341  tac_image_->SetSpacing(spacing);
342  typename Image4DType::PointType origin;
343  for(auto i=1; i<4; i++) origin[i] = images_[0]->GetOrigin()[i-1];
344  origin[0] = 0.0;
345  tac_image_->SetOrigin(origin);
346  tac_image_->Allocate();
347 
348  // Copy data to the 4D image
349  Iterator4D it(tac_image_, tac_image_->GetLargestPossibleRegion());
350  std::vector<Iterator3D> iterators;
351  for(auto image:images_) {
352  iterators.push_back(Iterator3D(image, image->GetLargestPossibleRegion()));
353  }
354  for(auto & iter:iterators) iter.GoToBegin();
355  for (it.GoToBegin(); !it.IsAtEnd(); ) {
356  for(auto & iter:iterators) {
357  it.Set(iter.Get());
358  ++iter;
359  ++it;
360  }
361  }
362 }
