40 int main(int argc, char* argv[])
41 {
42  // Init
43  SYD_INIT_GGO(syd_elastix, 1);
44 
45  // Load plugin
46  syd::PluginManager::GetInstance()->Load();
47  syd::DatabaseManager* m = syd::DatabaseManager::GetInstance();
48 
49  // Get the database
50  syd::StandardDatabase * db = m->Read<syd::StandardDatabase>(args_info.db_arg);
51 
52  // Get the elastix config file
53  std::string config_file = args_info.inputs[0];
54 
55  // Read ids from the command line and the pipe
56  std::vector<syd::IdType> ids;
57  syd::ReadIdsFromInputPipe(ids); // Read the standard input if pipe
58  for(auto i=1; i<args_info.inputs_num; i++)
59  ids.push_back(atoi(args_info.inputs[i]));
60  if (ids.size() < 2) {
61  LOG(FATAL) << "Please provide at least two image ids (fixed and moving images).";
62  }
63 
64  // Get the reference image
65  syd::Image::pointer fixed_image;
66  db->QueryOne(fixed_image, ids[0]);
67 
68  // Get the masks
69  syd::RoiMaskImage::pointer fixed_mask;
70  std::string fixed_mask_path;
71  if (args_info.fMask_given) {
72  fixed_mask = FindRoiMaskImage(db, fixed_image->patient, args_info.fMask_arg, fixed_image->frame_of_reference_uid);
73  fixed_mask_path = db->GetAbsolutePath(fixed_mask);
74  }
75 
76  // Loop over all images
77  for(auto i=1; i<ids.size(); i++) {
78 
79  syd::Image::pointer moving_image;
80  db->QueryOne(moving_image, ids[i]);
81 
82  syd::RoiMaskImage::pointer moving_mask;
83  std::string moving_mask_path;
84  if (args_info.mMask_given) {
85  moving_mask = FindRoiMaskImage(db, moving_image->patient, args_info.mMask_arg, moving_image->frame_of_reference_uid);
86  moving_mask_path = db->GetAbsolutePath(moving_mask);
87  }
88 
89  // Get elastix param
90  std::string options = args_info.options_arg;
91 
92  // Prepare the ImageTransform to get an id
93  syd::ImageTransform::pointer transfo;
94  db->New(transfo);
95  transfo->fixed_image = fixed_image;
96  transfo->moving_image = moving_image;
97  if (args_info.fMask_given) transfo->fixed_mask = fixed_mask;
98  if (args_info.mMask_given) transfo->moving_mask = moving_mask;
99  db->Insert(transfo); // insert to get id
100 
101  std::string f = syd::GetFilenameFromPath(config_file);
102  std::string output_dir = transfo->ComputeRelativeFolder();
103  bool b = fs::create_directories(db->ConvertToAbsolutePath(output_dir));
104  if (!b) {
105  LOG(FATAL) << "Error while creating " << output_dir;
106  }
107 
108  transfo->config_file = db->InsertNewFile(config_file, f, output_dir, true); // copy
109  transfo->transform_file = db->InsertNewFile("", "TransformParameters.0.txt", output_dir, false); // do not copy
110 
111  namespace pt = boost::posix_time;
112  std::ostringstream msg;
113  const pt::ptime now = pt::second_clock::local_time();
114  pt::time_facet*const ff = new pt::time_facet("%Y-%m-%d %H:%M:%S");
115  msg.imbue(std::locale(msg.getloc(),ff));
116  msg << now;
117  transfo->date = msg.str();
118 
119  // path
120  std::string fixed_image_path = db->GetAbsolutePath(fixed_image);
121  std::string moving_image_path = db->GetAbsolutePath(moving_image);
122 
123  // Create command line
124  std::ostringstream cmd;
125  cmd << "elastix -f " << fixed_image_path
126  << " -m " << moving_image_path
127  << " -out " << db->ConvertToAbsolutePath(output_dir)
128  << " -p " << config_file;
129  // masks
130  if (fixed_mask != NULL) cmd << " -fMask " << fixed_mask_path;
131  if (moving_mask != NULL) cmd << " -mMask " << moving_mask_path;
132  cmd << options; // additional options to elastix
133  std::ofstream os(output_dir+PATH_SEPARATOR+"command.txt");
134  os << msg.str() << std::endl
135  << fixed_image->id << " " << fixed_image_path << std::endl
136  << moving_image->id << " " << moving_image_path << std::endl
137  << config_file << std::endl
138  << cmd.str();
139  os.close();
140 
141  // Execute elastix
142  LOG(1) << cmd.str();
143  int r = syd::ExecuteCommandLine(cmd.str(), args_info.verbose_arg);
144 
145  if (r!=0) { // fail
146  LOG(WARNING) << "Command elastix fail, removing temporary folder and table element";
147  db->Delete(transfo);
148  }
149  else {
150  std::string res = db->GetAbsolutePath(transfo->transform_file);
151  if (!fs::exists(res)) {
152  LOG(WARNING) << "Error could not find the file, delete temporary folder and table element" << res;
153  db->Delete(transfo);
154  }
155  else { // only create the files if ok
156  db->Update(transfo);
157  LOG(1) << "Registration computed. Result: " << transfo;
158  }
159  }
160  }
161 
162  // This is the end, my friend.
163 }

169 syd::RoiMaskImage::pointer
170 FindRoiMaskImage(const syd::StandardDatabase * db,
171  const syd::Patient::pointer patient,
172  const std::string & mask_name_or_id,
173  const std::string & frame_of_reference_uid)
174 {
175  syd::RoiMaskImage::pointer mask;
176  // By id
177  syd::IdType id = atoi(mask_name_or_id.c_str());
178  if (id != 0) {
179  db->QueryOne(mask, id);
180  return mask;
181  }
182  // by roi name
183  syd::RoiType::pointer roitype = db->FindRoiType(mask_name_or_id);
184  syd::RoiMaskImage::vector temp;
185  odb::query<syd::RoiMaskImage> q =
186  odb::query<syd::RoiMaskImage>::patient == patient->id and
187  odb::query<syd::RoiMaskImage>::roitype == roitype->id and
188  odb::query<syd::RoiMaskImage>::frame_of_reference_uid == frame_of_reference_uid;
189  db->Query(temp, q);
190  if (temp.size() == 0) {
191  LOG(FATAL) << "No mask '" << roitype->name
192  << "' with frame_of_reference_uid "
193  << frame_of_reference_uid;
194  }
195  if (temp.size() > 1) {
196  std::string s;
197  for(auto t:temp) s += t->ToString()+" ";
198  LOG(FATAL) << "Several masks '" << roitype->name
199  << "' with frame_of_reference_uid found "
200  << frame_of_reference_uid << ": " << std::endl << s;
201  }
202  return temp[0];
203 }
