35 syd::Database * syd::DatabaseManager::Open(std::string filename)
36 {
37  odb::sqlite::database * db;
38  try {
39  LOG(5) << "Opening database '" << filename << "' to get typename.";
40  db = new odb::sqlite::database(filename);
41  odb::connection_ptr c(db->connection());
42  }
43  catch (const odb::exception& e) {
44  LOG(FATAL) << "Cannot open db '" << filename << "' : " << e.what();
45  }
46 
47  std::string dbtypename;
48  try {
49  // Try to get the tabletype name
50  odb::transaction transaction (db->begin());
51  odb::query<syd::DatabaseInformation> q;
52  typedef odb::result<syd::DatabaseInformation> result;
53  result r(db->query< syd::DatabaseInformation >(q));
54  syd::DatabaseInformation s;
55  if (r.begin() == r.end()) {
56  LOG(FATAL) << "Cannot find schema_version name in the db ?";
57  }
58  // We only consider the *last* element here. The table
59  // schema_version can contains several schema names, but only the
60  // last defined the database type.
61  for(auto i=r.begin(); i != r.end(); i++) {
62  r.begin().load(s);
63  }
64  transaction.commit();
65  dbtypename = s.database_type;
66  LOG(5) << "Database type is '" << dbtypename << "'.";
67  }
68  catch (const odb::exception& e) {
69  LOG(FATAL) << "Cannot find the db typename in '" << filename << "'."
70  << std::endl << "A table name 'schema_version' must be in the db. "
71  << "odb error is: " << e.what();
72  }
73 
74  auto it = db_map_.find(dbtypename);
75  if (it == db_map_.end()) {
76  LOG(FATAL) << "The database type '" << dbtypename << "' is not found."
77  << std::endl
78  << "Try to insert this typename via plugin.";
79  }
80  return it->second->Open(filename);
81 }
