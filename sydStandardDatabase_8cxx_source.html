140 void syd::StandardDatabase::FindTags(syd::Tag::vector & tags, const std::string & names) const
141 {
142  std::vector<std::string> words;
143  syd::GetWords(words, names);
144  odb::query<Tag> q = odb::query<Tag>::label.in_range(words.begin(), words.end());
145  Query<Tag>(tags, q);
146  if (tags.size() != words.size()) {
147  std::string s;
148  for(auto t:tags) s += t->ToString()+" ";
149  EXCEPTION("Cannot find all tags in FindTags: " << names << std::endl
