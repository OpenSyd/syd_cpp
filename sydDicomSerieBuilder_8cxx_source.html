66 void syd::DicomSerieBuilder::SetInjection(std::string p)
67 {
68  if (!db_) {
69  LOG(FATAL) << "Set database pointer before using SetInjection;";
70  }
71  int id = atoi(p.c_str());
72  if (db_->IfExist<Injection>(id)) {
73  injection_ = db_->QueryOne<Injection>(id);
74  }
75  else {
76  // Try with name
77  if (patient_.id != -1) {
78  std::vector<Injection> injections;
79  db_->Query<Injection>(odb::query<Injection>::radionuclide == p and
80  odb::query<Injection>::patient->name == patient_.name,
81  injections);
82  if (injections.size() == 1) injection_ = injections[0];
83  if (injections.size() > 1) {
84  LOG(FATAL) << "Several injection match '" << p << "' for the patient '" << patient_.name << "', abort.";
85  }
86  if (injections.size() == 0) {
87  LOG(FATAL) << "I cannot find an injection matching '" << p << "' for the patient '" << patient_.name << "', abort.";
88  }
89  }
90  else {
91  LOG(FATAL) << "sydDicomSerieBuilder : use SetPatient before SetInjection";
92  }
93  }
94 }

107 void syd::DicomSerieBuilder::SearchForFilesInFolder(OFList<OFString> & inputFiles)
108 {
109  // Search for all the files in the directory
110  inputFiles.clear();
111  std::string absolute_folder = folder_;
112  ConvertToAbsolutePath(absolute_folder);
113  LOG(2) << "Search for files in " << folder_;
114 
115  OFString scanPattern = "*"; // or *dcm ?
116  OFString dirPrefix = "";
117  OFBool recurse = OFTrue;
118  size_t found=0;
119  if (syd::DirExists(absolute_folder)) {
120  found = OFStandard::searchDirectoryRecursively(absolute_folder.c_str(),
121  inputFiles, scanPattern,
122  dirPrefix, recurse);
123  }
124  else {
125  LOG(FATAL) << "The directory " << absolute_folder << " does not exist.";
126  }
127  // I dont know why but sometimes, the recursive search find the same
128  // files several times. Here is a workaournd to remove duplicate
129  // files. To remove duplicate : conversion to set.
130  int n_before = inputFiles.size();
131  std::vector<OFString> v;
132  for(auto i=inputFiles.begin(); i!=inputFiles.end(); i++) v.push_back(*i);
133  std::set<OFString> s (v.begin(), v.end() );
134  v.assign( s.begin(), s.end() );
135  inputFiles.clear();
136  for(auto i=v.begin(); i<v.end(); i++) inputFiles.push_back(*i);
137  int n_after = inputFiles.size();
138  if (n_before != n_after) {
139  LOG(WARNING) << "Found duplicated files, I ignore them.";
140  }
141 
142  if (inputFiles.size() > 0) {
143  LOG(2) << "Found " << inputFiles.size() << " files. Now searching for dicom.";
144  }
145  else {
146  LOG(WARNING) << "No files found.";
147  }
148 }

153 void syd::DicomSerieBuilder::InsertDicomFile(std::string filename)
154 {
155  // Open the file
156  DcmFileFormat dfile;
157  bool b = syd::OpenDicomFile(filename, dfile);
158  if (!b) { // this is not a dicom file
159  LOG(WARNING) << "Error the file '" << filename << "' is not a dicom file.";
160  return;
161  }
162  DcmObject * dset = dfile.getAndRemoveDataset();
163 
164  // Test if this dicom file already exist
165  std::string sop_uid = GetTagValueString(dset, "SOPInstanceUID");
166  if (DicomFileAlreadyExist(sop_uid)) {
167  LOG(2) << "Dicom file with same sop_uid already exist in the db. Skipping " << filename;
168  nb_of_skip_files++;
169  return;
170  }
171 
172  // Test if a serie already exist in the database
173  DicomSerie * serie;
174  b = GuessDicomSerieForThisFile(filename, dset, &serie);
175 
176  // If this is a new DicomSerie, we create it
177  if (!b) {
178  serie = CreateDicomSerie(filename, dset);
179  series_to_insert.push_back(serie);
180  LOG(2) << "Creating a new serie: " << serie->dicom_series_uid;
181  }
182  // DD(*serie);
183 
184  // Then we add this dicomfile to the serie
185  DicomFile * dicomfile = CreateDicomFile(filename, dset, serie);
186 
187  // FIXME Check
188  for(auto d:dicomfiles_to_insert) {
189  if (d->dicom_sop_uid == dicomfile->dicom_sop_uid) {
190  LOG(FATAL) << "FOUND duplicate !!";

193 
194  dicomfiles_to_insert.push_back(dicomfile);

213 bool syd::DicomSerieBuilder::GuessDicomSerieForThisFile(const std::string & filename,
214  DcmObject * dset,
215  syd::DicomSerie ** serie)
216 {
217  // Check in the future series
218  std::string series_uid = GetTagValueString(dset, "SeriesInstanceUID");
219  std::string modality = GetTagValueString(dset, "Modality");
220  int index = -1;
221  for(auto i=0; i<series_to_insert.size(); i++) {
222  DicomSerie * s = series_to_insert[i];
223  // First check series_uid and modality
224  if (s->dicom_series_uid != series_uid) continue;
225  if (modality != s->dicom_modality) continue;
226 
227  // Here we found a DicomSerie with the same series_uid
228  // Very simple heuristic based on the modality
229  if (s->dicom_modality == "CT") {
230  if (index != -1) {
231  LOG(FATAL) << "Error two different CT DicomSerie with the same series_uid exist. Database corrupted."
232  << std::endl
233  << "First serie: " << series_to_insert[index] << std::endl
234  << "Second serie: " << series_to_insert[i] << std::endl;
235  }
236  index = i;
237  }
238  }
239  // We find a corresponding serie
240  if (index != -1) {
241  *serie = series_to_insert[index];
242  return true;
243  }
244 
245  // Find all existing DicomSerie with the same uid, in the db
246  std::vector<DicomSerie> series;
247  db_->Query<DicomSerie>(odb::query<DicomSerie>::dicom_series_uid == series_uid, series);
248  index=-1;
249  for(auto i=0; i<series.size(); i++) {
250  DicomSerie * s = &series[i];
251  if (modality != s->dicom_modality) continue;
252  // Here we found a DicomSerie with the same series_uid
253  // Very simple heuristic based on the modality
254  if (s->dicom_modality == "CT") {
255  if (index != -1) {
256  LOG(FATAL) << "Error two different CT DicomSerie with the same series_uid exist. Database corrupted."
257  << std::endl
258  << "First serie: " << series[index] << std::endl
259  << "Second serie: " << s << std::endl;
260  }
261  index = i;
262  }
263  }
264 
265  // We find a corresponding serie
266  if (index != -1) {
267  *serie = &series[index];
268  return true;
269  }
270 
271  // Nothing found
272  return false;
273 }

290 void syd::DicomSerieBuilder::UpdateDicomSerie(DicomSerie * serie,
291  const std::string & filename,
292  DcmObject * dset)
293 {
294  // Patient, injection (do not check here that injection is really associated with the patient)
295  std::string patientID = GetTagValueString(dset, "PatientID");
296  std::string patientName = GetTagValueString(dset, "PatientName");
297  LOG(2) << "Check patient dicom_patientid is the same than the given patient";
298  bool b = patient_.CheckIdentity(patientID, patientName);
299  if (!b and !forcePatientFlag_) {
300  LOG(FATAL) << "Patient do not seems to be the same. You ask for " << patient_.name
301  << " with dicom_id = '" << patient_.dicom_patientid << "'"
302  << " while in dicom, it is '" << patientID << "'" << std::endl
303  << "Use 'forcePatient' if you want to bypass this check";
304  }
305  if (!b) {
306  // if the patient has no dicom id, we set it (and update the db)
307  patient_.dicom_patientid = patientID;
308  LOG(2) << "The dicom_id of the patient " << patient_ << " has been updated to " << patientID;
309  db_->Update<Patient>(patient_);
310  }
311  serie->patient = std::make_shared<syd::Patient>(patient_);
312  if (useInjectionFlag_)
313  serie->injection = std::make_shared<syd::Injection>(injection_);
314 
315  // Modality
316  serie->dicom_modality = GetTagValueString(dset, "Modality");
317 
318  // UID
319  serie->dicom_study_uid = GetTagValueString(dset, "StudyInstanceUID");
320  serie->dicom_series_uid = GetTagValueString(dset, "SeriesInstanceUID");
321  serie->dicom_frame_of_reference_uid = GetTagValueString(dset, "FrameOfReferenceUID");
322  // serie->dicom_dataset_uid = GetTagValueString(dset, "DatasetUID");
323 
324  // Date
325  std::string AcquisitionTime = GetTagValueString(dset, "AcquisitionTime");
326  std::string AcquisitionDate = GetTagValueString(dset, "AcquisitionDate");
327  std::string ContentDate = GetTagValueString(dset, "ContentDate");
328  std::string ContentTime = GetTagValueString(dset, "ContentTime");
329  std::string InstanceCreationDate = GetTagValueString(dset, "InstanceCreationDate");
330  std::string InstanceCreationTime = GetTagValueString(dset, "InstanceCreationTime");
331  std::string acquisition_date = ConvertDicomDateToStringDate(AcquisitionDate, AcquisitionTime);
332  std::string reconstruction_date = ConvertDicomDateToStringDate(ContentDate, ContentTime);
333  if (reconstruction_date.empty())
334  reconstruction_date = ConvertDicomDateToStringDate(InstanceCreationDate, InstanceCreationTime);
335  serie->acquisition_date = acquisition_date;
336  serie->reconstruction_date = reconstruction_date;
337 
338  // Description. We merge the tag because it is never consistant
339  std::string SeriesDescription = GetTagValueString(dset, "SeriesDescription");
340  std::string StudyDescription = GetTagValueString(dset, "StudyDescription");
341  std::string ImageID = GetTagValueString(dset, "ImageID");
342  std::string DatasetName = GetTagValueString(dset, "DatasetName");
343  std::string description = SeriesDescription+" "+StudyDescription
344  +" "+ImageID+" "+DatasetName;
345  serie->dicom_description = description;
346 
347  // Device
348  std::string Manufacturer = GetTagValueString(dset, "Manufacturer");
349  std::string ManufacturerModelName = GetTagValueString(dset, "ManufacturerModelName");
350  std::string device=Manufacturer+" "+ManufacturerModelName;
351  serie->dicom_manufacturer = device;
352 
353  // Image size
354  int rows = GetTagValueUShort(dset, "Rows");
355  int columns = GetTagValueUShort(dset, "Columns");
356  serie->size[0] = columns;
357  serie->size[1] = rows;
358  serie->size[2] = 0;
359 
360  int slice = atoi(GetTagValueString(dset, "NumberOfFrames").c_str());
361  serie->size[2] = slice;
362 
363  // other (needed ?)
364  // std::string TableTraverse = GetTagValueString(dset, "TableTraverse");
365  // std::string InstanceNumber = GetTagValueString(dset, "InstanceNumber");
366 }

372 syd::DicomFile * syd::DicomSerieBuilder::CreateDicomFile(const std::string & filename,
373  DcmObject * dset,
374  DicomSerie * serie)
375 {
376  // First create the file
377  syd::File * file = new syd::File;
378  std::string f = GetFilenameFromPath(filename);
379  file->filename = f;
380  std::string p = db_->GetFolder(*serie);
381  std::string rp = p;
382  syd::Replace(rp, db_->GetAbsoluteDBFolder()+PATH_SEPARATOR, ""); // Get relative folder
383  file->path = rp;
384  files_to_copy.push_back(filename);
385  destination_folders.push_back(p);
386  // file->md5 FIXME later
387 
388  // Then create the dicomfile
389  syd::DicomFile * dicomfile = new syd::DicomFile;
390  dicomfile->file = std::shared_ptr<syd::File>(file);
391  dicomfile->dicom_serie = std::shared_ptr<syd::DicomSerie>(serie);
392  std::string sop_uid = GetTagValueString(dset, "SOPInstanceUID");
393  dicomfile->dicom_sop_uid = sop_uid;
394  int instance_number = atoi(GetTagValueString(dset, "InstanceNumber").c_str());
395  dicomfile->dicom_instance_number = instance_number;
396 
397  // Update the nb of slices
398  serie->size[2]++;
399 
400  return dicomfile;
401 }

406 void syd::DicomSerieBuilder::UpdateSeries()
407 {
408  // Gather the syd::Files to update
409  std::vector<syd::File*> files;
410  for(auto d:dicomfiles_to_insert) {
411  files.push_back(d->file.get());
412  }
413  db_->Insert(files);
414 
415  // Copy files
416  int nb_of_skip_copy=0;
417  int n = files_to_copy.size();
418  for(auto i=0; i<files_to_copy.size(); i++) {
419  std::string f = syd::GetFilenameFromPath(files_to_copy[i]);
420  std::string destination = destination_folders[i]+PATH_SEPARATOR+f;
421  if (syd::FileExists(destination)) {
422  LOG(3) << "Destination file already exist, ignoring";
423  nb_of_skip_copy++;
424  continue;
425  }
426  LOG(2) << "Copying " << f << " to " << destination_folders[i] << std::endl;
427  syd::CopyFile(files_to_copy[i].c_str(), destination);
428  // std::ifstream src(files_to_copy[i].c_str(), std::ios::binary);
429  // std::ofstream dst(destination, std::ios::binary);
430  // dst << src.rdbuf();
431  syd::loadbar(i,n);
432  }
433 
434  // Update the database
435  db_->Insert(series_to_insert);
436  db_->Insert(dicomfiles_to_insert);
437 
438  // Log
439  LOG(1) << files.size() << " Files has been added in the db";
440  LOG(1) << dicomfiles_to_insert.size() << " DicomFiles has been added in the db";
441  LOG(1) << series_to_insert.size() << " DicomSeries has been added in the db";
442  if (nb_of_skip_files != 0) {
443  LOG(1) << nb_of_skip_files << " dicom already exist in the db and have been skipped.";
444  }
445  if (nb_of_skip_copy != 0) {
446  LOG(1) << nb_of_skip_copy << " files already exist in the db folder and have not been copied.";
447  }
448 
449  // Once done, clear vectors
450  series_to_insert.clear();
451  dicomfiles_to_insert.clear();
452  files.clear();
453  files_to_copy.clear();
454  destination_folders.clear();
455  nb_of_skip_files = 0;
456 }
