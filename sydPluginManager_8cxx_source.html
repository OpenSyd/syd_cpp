49 void syd::PluginManager::LoadInFolder(std::string folder)
50 {
51  std::string absolute_folder = folder;
52  ConvertToAbsolutePath(absolute_folder);
53  OFString scanPattern = "*"; // or *dcm ?
54  OFString dirPrefix = "";
55  OFBool recurse = OFFalse;
56  size_t found=0;
57  OFList<OFString> inputFiles;
58  if (syd::DirExists(absolute_folder)) {
59  found = OFStandard::searchDirectoryRecursively(absolute_folder.c_str(),
60  inputFiles, scanPattern,
61  dirPrefix, recurse);
62  }
63  else {
64  LOG(FATAL) << "(plugin) The directory " << absolute_folder << " does not exist.";
65  }
66 
67  for(auto f=inputFiles.begin(); f != inputFiles.end(); f++) {
68  Load(f->c_str());
69  }
70 }

75 void syd::PluginManager::Load(std::string filename)
76 {
77  std::string ext = GetExtension(filename);
78  if (ext != "so" and ext != "dylib" and ext != "DLL" and ext != "dll") {
79  LOG(10) << "(plugin) ignoring the file '" << filename << "'.";
80  return;
81  }
82  void * plugin;
83  plugin = dlopen(filename.c_str(), RTLD_NOW);
84  if (!plugin) {
85  LOG(10) << "(plugin) The file '" << filename << "' is not a plugin.";
86  return;
87  }
88  RegisterDatabaseTypeFunction db_creator;
89  db_creator = (RegisterDatabaseTypeFunction) dlsym(plugin, "RegisterDatabaseType");
90  char * result = dlerror();
91  if (result) {
92  LOG(10) << "(plugin) The lib " << filename << " does not contain a valid plugin for syd.";
93  return;
94  }
95  syd::DatabaseManager * m = syd::DatabaseManager::GetInstance();
96  db_creator(m);
97 }
