52 void syd::TimepointBuilder::InsertDicomSerie(syd::DicomSerie & dicom)
53 {
54  // Consider the other timepoint for this injection/tag
55  if (dicom.injection == NULL) {
56  LOG(WARNING) << "Error no associated injection with the dicom: " << dicom << ", ignoring.";
57  return;
58  }
59 
60  // Guess
61  bool isNew=false;
62  Timepoint tp;
63  GuessState b=GuessTimepointForThisDicomSerie(dicom, tp);
64  if (b == GuessError) return;
65  if (b == GuessNotFound) {
66  isNew = true;
67  tp = CreateTimepoint(dicom);
68  db_->Insert(tp);
69  }
70  AddDicom(tp, dicom);
71  db_->Update(tp);
72  if (isNew) {
73  LOG(1) << "Create new timepoint: " << tp;
74  }
75  else {
76  LOG(1) << "Add dicom to existing timepoint: " << tp;
77  }
78 }

83 syd::TimepointBuilder::GuessState
84 syd::TimepointBuilder::GuessTimepointForThisDicomSerie(syd::DicomSerie & dicom, syd::Timepoint & tp)
85 {
86  syd::Injection injection = *dicom.injection;
87  std::vector<syd::Timepoint> timepoints;
88  db_->Query<syd::Timepoint>(odb::query<syd::Timepoint>::injection->id == injection.id and
89  odb::query<syd::Timepoint>::tag->id == tag_.id, timepoints);
90  int n=0;
91  std::vector<syd::Timepoint> a;
92  for(auto t:timepoints) {
93  GuessState b = GuessIfDicomCanBeInThisTimepoint(dicom, t);
94  if (b == GuessFound) { ++n; a.push_back(t); }
95  if (b == GuessError) return b;
96  }
97  if (n==0) return GuessNotFound;
98  if (n==1) { tp = a[0]; return GuessFound; }
99  LOG(WARNING) << "Error, this dicom could belong to several timepoints, something is wrong." << std::endl
100  << "Dicom: " << dicom << std::endl
101  << "Timepoint 0: " << a[0] << std::endl
102  << "Timepoint 1: " << a[1] << std::endl;
103  return GuessError;
104 }

109 syd::TimepointBuilder::GuessState
110 syd::TimepointBuilder::GuessIfDicomCanBeInThisTimepoint(syd::DicomSerie & dicom, syd::Timepoint & timepoint)
111 {
112  if (timepoint.dicoms.size() == 0) {
113  LOG(FATAL) << "Error this timepoint has no associated dicom ?" << std::endl
114  << "Timepoint: " << timepoint;
115  }
116  for(auto d:timepoint.dicoms) {
117  double diff = fabs(syd::DateDifferenceInHours(dicom.acquisition_date, d->acquisition_date));
118  if (diff < intra_timepoint_min_hour_diff_) {
119  if (dicom.id == d->id) {
120  LOG(WARNING) << "The dicom is already inside the following timpoint:" << timepoint
121  << std::endl << "Dicom: " << dicom;
122  return GuessError;
123  }
124  LOG(WARNING) << "The dates are too close for this dicom and the dicoms of this timepoint." << std::endl
125  << "Dicom: " << dicom << std::endl
126  << "Timepoint: " << timepoint << std::endl
127  << " with dicom: " << *d << std::endl;
128  return GuessError;
129  }
130  if (diff > intra_timepoint_max_hour_diff_) return GuessNotFound;
131  }
132  return GuessFound;
133 }

154 void syd::TimepointBuilder::AddDicom(syd::Timepoint & timepoint, syd::DicomSerie & dicom)
155 {
156  timepoint.dicoms.push_back(std::make_shared<syd::DicomSerie>(dicom));
157 
158  // policy if several dicom: consider the first one
159  std::string date = timepoint.dicoms[0]->acquisition_date;
160  for(auto d:timepoint.dicoms) {
161  double diff = syd::DateDifferenceInHours(date, d->acquisition_date);
162  if (diff>0) date = d->acquisition_date;
163  }
164  timepoint.time_from_injection_in_hours =
165  syd::DateDifferenceInHours(date, timepoint.injection->date);
166 }
