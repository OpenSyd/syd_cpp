41 void syd::Database::Read(std::string filename)
42 {
43  filename_ = filename;
44 
45  // Open the DB
46  try {
47  LOG(4) << "Opening database '" << filename_ << "'.";
48  db_ = new odb::sqlite::database(filename_);
49  odb::connection_ptr c(db_->connection());
50  c->execute("PRAGMA foreign_keys=ON;");
51  }
52  catch (const odb::exception& e) {
53  LOG(FATAL) << "Cannot open db '" << filename_ << "' : " << e.what();
54  }
55 
56  // Set the folder by reading the 'folder' value in the db_info table.
57  try {
58  odb::transaction transaction (db_->begin());
59  typedef odb::query<syd::DatabaseInformation> query;
60  typedef odb::result<syd::DatabaseInformation> result;
61  query q;
62  result r (db_->query<syd::DatabaseInformation>(q));
63  syd::DatabaseInformation s;
64  r.begin().load(s);
65  relative_folder_ = s.folder;
66  database_schema_ = s.database_schema;
67  transaction.commit();
68  }
69  catch (const odb::exception& e) {
70  LOG(FATAL) << "Could not get the folder name ?" << e.what();
71  }
72 
73  // Now consider the folder according to the filename path and check it exists
74  std::string pwd;
75  if (!syd::GetWorkingDirectory(pwd)) {
76  LOG(FATAL) << "Error while trying to get current working dir.";
77  }
78  if (filename_[0] != PATH_SEPARATOR) {
79  pwd = pwd+PATH_SEPARATOR+filename_;
80  unsigned found = pwd.find_last_of(PATH_SEPARATOR);
81  pwd = pwd.substr(0,found);
82  }
83  absolute_folder_ = pwd+PATH_SEPARATOR+relative_folder_;
84  if (!syd::DirExists(absolute_folder_)) {
85  LOG(FATAL) << "The folder '" << absolute_folder_ << "' does not exist.";
86  }
87 
88  // Install tracer
89  odb::sqlite::connection_ptr c (db_->connection ());
90  sqlite3* handle (c->handle ());
91  sqlite3_trace (handle, trace_callback, this);
92 
93  // Define the tables
94  CreateTables();
95 }

119 void syd::Database::Dump(const std::vector<std::string> & args, std::ostream & os)
120 {
121  if (args.size() == 0) {
122  os << "Database schema: " << GetDatabaseSchema() << std::endl;
123  os << "Database folder: " << GetRelativeDBFolder() << std::endl;
124  for(auto i=map.begin(); i != map.end(); i++) {
125  int n = i->second->GetNumberOfElements();
126  os << "Table \t" << std::setw(15) << i->first << " " << std::setw(10) << n;
127  if (n>1) os << " elements" << std::endl;
128  else os << " element" << std::endl;
129  }
130  os << std::flush;
131  return;
132  }
133  if (args[0] == "table") {
134  if (args.size() < 2) {
135  LOG(FATAL) << "Please provide the name of the table you want to dump.";
136  }
137  DumpTable(args[1], os);
138  return;
139  }
140  DumpTable(args[0], os); // suppose it is a table name
141 }
