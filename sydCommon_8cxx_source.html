194 void syd::loadbar(unsigned int x, unsigned int n, unsigned int w)
195 {
196  // First check the last value to break line.
197  if (x == n-1) {
198  std::cout << std::endl;
199  }
200 
201  // Do nothing if not a %
202  if ( (x != n) && (x % (n/100+1) != 0) ) return;
203 
204  float ratio = x/(float)n;
205  int c = ratio * w;
206 
207  std::cout << std::setw(3) << (int)(ratio*100) << "% [";
208  for (int x=0; x<c; x++) std::cout << "=";
209  for (int x=c; x<w; x++) std::cout << " ";
210  std::cout << "]\r" << std::flush;
211 }

268 int syd::ExecuteCommandLine(const std::string & cmd, int logLevel)
269 {
270  const redi::pstreams::pmode mode = redi::pstreams::pstdout|redi::pstreams::pstderr;
271  redi::ipstream child(cmd, mode);
272  char buf[1024];
273  std::streamsize n;
274  bool finished[2] = { false, false };
275  const std::string color = "\x1b[32m"; // green
276  std::cout << color;
277  bool error = false;
278  while (!finished[0] || !finished[1]) {
279  if (!finished[0]) {
280  while ((n = child.err().readsome(buf, sizeof(buf))) > 0) {
281  error = true;
282  std::cerr << fatalColor;
283  std::cerr.write(buf, n);
284  }
285  if (child.eof()) {
286  finished[0] = true;
287  if (!finished[1]) child.clear();
288  }
289  }
290 
291  if (!finished[1]) {
292  while ((n = child.out().readsome(buf, sizeof(buf))) > 0) {
293  if (Log::LogLevel() >= logLevel) std::cout.write(buf, n).flush(); // only print if level ok
294  }
295  if (child.eof()) {
296  finished[1] = true;
297  if (!finished[0]) child.clear();
298  }
299  }
300  }
301  std::cout << DD_RESET;
302  if (error) return -1; // Error
303  return 0; // ok
304 }
