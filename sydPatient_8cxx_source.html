128 bool syd::Patient::CheckIdentity(std::string vdicom_patientid, std::string vdicom_name) const {
129  if (dicom_patientid != vdicom_patientid) return false;
130  // Try to guess initials. Consider the first letter and the first after the symbol '^'
131  int n = vdicom_name.find("^");
132  if (n != std::string::npos) {
133  std::string initials = vdicom_name[0] + vdicom_name.substr(n+1,1);
134  std::transform(initials.begin(), initials.end(), initials.begin(), ::tolower);
135  // Check only 2 first letters
136  if (initials[0] == name[0] and initials[1] == name[1]) return true;
137  return false;
138  }
139  return false; // do not consider ok with a patient name not having a '^' inside
140 }
