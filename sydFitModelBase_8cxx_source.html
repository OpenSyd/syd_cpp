81 double syd::FitModelBase::Integrate(double a, double b, double l_phys) const

109 double syd::FitModelBase::ComputeAUC(const syd::TimeActivityCurve & tac, double l_phys, bool use_current_tac) const
110 {
111  // Simple integration if full model
112  if (!use_current_tac) return Integrate(l_phys);
113 
114  // If not, we consider the current_tac as the restricted one
115  if (!current_tac) {
116  LOG(FATAL) << "Could not compute ComputeAUC with restricted tac, 'current_tac' must be set";
117  }
118  double AUC = 0.0;
119 
120  // Integrate from 0 to first time of the restricted_tac
121  double starting_part_model = Integrate(0.0, current_tac->GetTime(0), l_phys);
122 
123  // Integrate from 0 to infinity
124  double total = Integrate(l_phys);
125 
126  // Trapeze intregration of the first curve part
127  int index = 0;
128  double t = current_tac->GetTime(0);
129  while (tac.GetTime(index) < t) ++index;
130  double starting_part = tac.Integrate_Trapeze(0, index);
131 
132  AUC = total - starting_part_model + starting_part;
133 
134  return AUC;
135 }

140 double syd::FitModelBase::ComputeR2(const syd::TimeActivityCurve & tac, bool use_current_tac) const
141 {
142  const syd::TimeActivityCurve * current = &tac;
143  if (use_current_tac) {
144  if (!current_tac) {
145  LOG(FATAL) << "Could not compute ComputeAUC with restricted tac, 'current_tac' must be set";
146  }
147  current = current_tac;
148  }
149 
150  double mean = 0.0;
151  for(auto i=0; i<current->size(); i++) mean += current->GetValue(i);
152  mean = mean / (double)current->size();
153 
154  double SS_res = ComputeSS(*current);
155 
156  double SS_tot = 0.0;
157  for(auto i=0; i<current->size(); i++) {
158  SS_tot += pow(current->GetValue(i)-mean, 2);
159  }
160 
161  double R2 = 1.0 - (SS_res/SS_tot);
162  return R2;
163 }

189 double syd::FitModelBase::ComputeAICc(const syd::TimeActivityCurve & tac) const
190 {
191  // See Glatting 2007 equ (4)
192  // See Kletting 2013 eq (5) / (6) and erratum
193  // See Burnham2011 and others
194  double N = tac.size();
195  double K = GetK();
196  double SS = ComputeSS(tac);
197 
198  double AIC = N * log(SS/N) + 2.0*(K+1) ; // std::log is ln (natural logarithm), least square fit with normally dist errors
199  //double AICc = AIC + (2*(K+1.0)*(K+2.0))/(N-K-2.0); // relative -> but issue if N-K-2 too low.
200  double AICc = AIC + (2*K*(K+1))/(N-K-1); // absolute
201 
202  if (K+2 > N) AICc = 666; // not applicable
203  // if (K+3 > N) AICc = 666; // not applicable
204 
205  // std::cout << std::endl << name_ << " N=" << N << " K=" << K << " SS= " << SS
206  // << " -> " << AIC << " " << AICc << std::endl;
207 
208  return AICc;
209 }
