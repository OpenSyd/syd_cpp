40 syd::RoiMaskImageBuilder::InsertRoiMaskImage(const syd::DicomSerie::pointer & dicom,
41  const syd::RoiType::pointer & roitype,
42  const std::string & filename)
43 {
44  syd::RoiMaskImage::pointer mask;
45  syd::Image::pointer image;
46  try {
47  // Create a mask
48  db_->New(mask);
49  mask->roitype = roitype;
50  mask->patient = dicom->patient;
51  mask->dicoms.push_back(dicom);
52  mask->frame_of_reference_uid = dicom->dicom_frame_of_reference_uid;
53  mask->type = "mhd";
54  syd::Tag::pointer tag_mask = db_->FindOrInsertTag("mask", "Mask image");
55  mask->AddTag(tag_mask);
56  db_->Insert(mask);
57 
58  // Create filename
59  std::ostringstream oss;
60  oss << roitype->name << "_" << mask->id << ".mhd";
61  std::string mhd_filename = oss.str();
62  std::string mhd_relative_path = mask->ComputeRelativeFolder()+PATH_SEPARATOR;
63  std::string mhd_path = db_->ConvertToAbsolutePath(mhd_relative_path+mhd_filename);
64 
65  // Update files in images
66  mask->UpdateFile(db_, mhd_filename, mhd_relative_path);
67 
68  // Read image
69  typedef unsigned char PixelType;
70  typedef itk::Image<PixelType,3> ImageType;
71  ImageType::Pointer itk_image;
72  itk_image = syd::ReadImage<ImageType>(filename);
73 
74  // Update image info
75  db_->UpdateImageInfo<PixelType>(mask, itk_image, true); // true = update md5
76  LOG(4) << "Write image on disk " << mhd_path;
77  syd::WriteImage<ImageType>(itk_image, mhd_path);
78 
79 
80  } catch (syd::Exception & e) {
81  EXCEPTION("Error during InsertImage: " << e.what());
82  }
83 
84  // Final update
85  db_->Update(mask);
86  return mask;
87 }
