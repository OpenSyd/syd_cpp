24 std::string syd::ConvertDicomSPECTFileToImage(std::string dicom_filename, std::string mhd_filename)
25 {
26  // Open the dicom
27  DcmFileFormat dfile;
28  bool b = syd::OpenDicomFile(dicom_filename.c_str(), dfile);
29  if (!b) {
30  LOG(FATAL) << "Could not open the dicom file '" << dicom_filename
31  << "' maybe this is not a dicom ?";
32  }
33  DcmObject *dset = dfile.getDataset();
34 
35  // Read the image data
36  LOG(2) << "Converting SPECT dicom to mhd (" << mhd_filename << ") ...";
37  typedef float PixelType;
38  typedef itk::Image<PixelType, 3> ImageType;
39  ImageType::Pointer image = ReadImage<ImageType>(dicom_filename);
40 
41  // Check nb of slices
42  typename ImageType::SizeType size = image->GetLargestPossibleRegion().GetSize();
43  ushort nbslices = GetTagValueUShort(dset, "NumberOfSlices");
44  if (nbslices != size[2]) {
45  LOG(FATAL) << "Error image spacing is " << size
46  << " while in the dicom NumberOfSlices = " << nbslices;
47  }
48 
49  // Remove meta information (if not : garbage in the mhd)
50  itk::MetaDataDictionary d;
51  image->SetMetaDataDictionary(d);
52 
53  // Offset
54  std::string ImagePositionPatient = GetTagValueString(dset, "ImagePositionPatient");
55  if (ImagePositionPatient == "") {
56  LOG(FATAL) << "Error while reading tag ImagePositionPatient in the dicom ";
57  }
58  int n = ImagePositionPatient.find("\\");
59  std::string sx = ImagePositionPatient.substr(0,n);
60  ImagePositionPatient = ImagePositionPatient.substr(n+1,ImagePositionPatient.size());
61  n = ImagePositionPatient.find("\\");
62  std::string sy = ImagePositionPatient.substr(0,n);
63  std::string sz = ImagePositionPatient.substr(n+1,ImagePositionPatient.size());
64  typename ImageType::PointType origin;
65  origin[0] = ToDouble(sx);
66  origin[1] = ToDouble(sy);
67  origin[2] = ToDouble(sz);
68  image->SetOrigin(origin);
69 
70  // Correct for negative SpacingBetweenSlices
71  double s = GetTagValueDouble(dset, "SpacingBetweenSlices");
72  // change spacing z
73  typename ImageType::SpacingType spacing = image->GetSpacing();
74  if (s<0) spacing[2] = -s;
75  else spacing[2] = s;
76  image->SetSpacing(spacing);
77  // Direction
78  if (s<0) {
79  typename ImageType::DirectionType direction = image->GetDirection();
80  direction.Fill(0.0);
81  direction(0,0) = 1; direction(1,1) = 1; direction(2,2) = -1;
82  image->SetDirection(direction);
83 
84  // Change orientation
85  itk::OrientImageFilter<ImageType,ImageType>::Pointer orienter =
86  itk::OrientImageFilter<ImageType,ImageType>::New();
87  orienter->UseImageDirectionOn();
88  orienter->SetDesiredCoordinateOrientation(itk::SpatialOrientation::ITK_COORDINATE_ORIENTATION_RAI);
89  orienter->SetInput(image);
90  orienter->Update();
91  image = orienter->GetOutput();
92  }
93 
94  // Correct for pixel scale --> dont know why it does not work.
95  // DcmDictEntry * e = new DcmDictEntry(0x0011, 0x103b, EVR_UN, "PixelScale", 0, DcmVariableVM, NULL, true, NULL);
96  // DcmDataDictionary &globalDataDict = dcmDataDict.wrlock();
97  // globalDataDict.addEntry(e);
98  // DcmElement * ee = GetTagValue(dset, "PixelScale");
99  // double f;
100  // ee->getFloat64(f);
101  // DD(f);
102  /*
103  double scale = GetTagValueDouble(dset, "PixelScale");
104  DD(scale);
105  if (scale != 1.0 and scale != 0.0) {
106  typedef itk::MultiplyImageFilter<ImageType, ImageType, ImageType> FilterType;
107  typename FilterType::Pointer filter = FilterType::New();
108  filter->SetInput(image);
109  filter->SetConstant(scale);

125 std::string syd::ConvertDicomCTFolderToImage(std::string dicom_path, std::string mhd_filename)
126 {
127  // read all dicom in the folder
128  typedef itk::Image<signed short, 3> ImageType;
129  typedef itk::ImageSeriesReader<ImageType> ReaderType;
130  typedef itk::GDCMImageIO ImageIOType;
131  typedef itk::GDCMSeriesFileNames InputNamesGeneratorType;
132  ImageIOType::Pointer gdcmIO = ImageIOType::New();
133  InputNamesGeneratorType::Pointer inputNames = InputNamesGeneratorType::New();
134  inputNames->SetInputDirectory(dicom_path);
135  const ReaderType::FileNamesContainer & filenames = inputNames->GetInputFileNames();
136  LOG(2) << "Converting CT dicom (with " << filenames.size() << " files) to mhd (" << mhd_filename << ") ...";
137  ReaderType::Pointer reader = ReaderType::New();
138  reader->SetImageIO( gdcmIO );
139  reader->SetFileNames( filenames );
140  try { reader->Update(); }
141  catch (itk::ExceptionObject &excp) {
142  std::cerr << excp << std::endl;
143  LOG(FATAL) << "Error while reading the dicom serie in " << dicom_path << " ";
144  }
145 
146  // convert to mhd and write
147  ImageType::Pointer ct = reader->GetOutput();
148  syd::WriteImage<ImageType>(ct, mhd_filename);
149 
150  // Compute md5
151  return ComputeImageMD5<ImageType>(ct);
152 }

158  Rename a mhd image, composed of 2 files XXX.mhd and XXX.raw. This
159  function moves the two files and change the header to be linked with
160  the renamed .raw file.

162 void syd::RenameOrCopyMHDImage(std::string old_path, std::string new_path, int verbose_level, bool erase)
163 {
164  // Check if not the same
165  if (old_path == new_path) {
166  LOG(WARNING) << "Try to rename " << old_path << " to the same file. Do nothing.";
167  return; // do nothing
168  }
169 
170  // Check extension old filename
171  size_t n = old_path.find_last_of(".");
172  std::string extension = old_path.substr(n+1);
173  if (extension != "mhd") {
174  LOG(FATAL) << "Rename MHD : Error the filename must have mhd as extension : " << old_path;
175  }
176  std::string old_path_raw = old_path.substr(0,n)+".raw";
177 
178  // Check extension new filename
179  n = new_path.find_last_of(".");
180  extension = new_path.substr(n+1);
181  if (extension != "mhd") {
182  LOG(FATAL) << "Rename MHD : Error the new filename must have mhd as extension : " << new_path;
183  }
184  std::string new_path_raw = new_path.substr(0,n)+".raw";
185 
186  // Check files
187  if (!syd::FileExists(old_path)) {
188  LOG(FATAL) << "Rename MHD : Error path (mhd) not exist : " << old_path;
189  }
190  if (!syd::FileExists(old_path_raw)) {
191  LOG(FATAL) << "Rename MHD : Error path (raw) not exist : " << old_path_raw;
192  }
193  if (syd::FileExists(new_path)) {
194  LOG(WARNING) << "Rename MHD : path (mhd) to rename already exist : " << new_path;
195  }
196  if (syd::FileExists(new_path_raw)) {
197  LOG(WARNING) << "Rename MHD : path (raw) to rename already exist : " << new_path_raw;
198  }
199 
200  // verbose
201  if (erase) {
202  LOG(verbose_level) << "Rename header " << old_path << " to " << new_path;
203  }
204  else {
205  LOG(verbose_level) << "Copy header " << old_path << " to " << new_path;
206  }
207 
208  // header part : change ElementDataFile in the header
209  std::ifstream in(old_path);
210  // std::ofstream out(new_path);
211  std::vector<std::string> outlines;
212  // std::ofstream out;
213  OFString r;
214  OFStandard::getFilenameFromPath(r, old_path_raw.c_str());
215  std::string wordToReplace(r.c_str());
216  OFStandard::getFilenameFromPath(r, new_path_raw.c_str());
217  std::string wordToReplaceWith(r.c_str());
218  size_t len = wordToReplace.length();
219  std::string line;
220  // Delete old path (before writing new)
221  if (erase) std::remove(old_path.c_str());
222  while (std::getline(in, line)) {
223  while (line != "") {
224  size_t pos = line.find(wordToReplace);
225  if (pos != std::string::npos) {
226  line.replace(pos, len, wordToReplaceWith);
227  break;
228  }
229  else
230  break;
231  }
232  outlines.push_back(line+"\n");
233  }
234  in.close();
235 
236  // Write
237  std::ofstream out(new_path);
238  for(auto i:outlines) out << i;
239  out.close();
240 
241  // Rename or copy .raw part
242  if (erase) {
243  LOG(verbose_level) << "Rename raw " << old_path_raw << " to " << new_path_raw;
244  int result = std::rename(old_path_raw.c_str(), new_path_raw.c_str());
245  if (result != 0) {
246  LOG(FATAL) << "Error while renaming " << old_path_raw << " to " << new_path_raw;
247  }
248  }
249  else {
250  LOG(verbose_level) << "Copy raw " << old_path_raw << " to " << new_path_raw;
251  // std::ifstream src(old_path_raw.c_str(), std::ios::binary);
252  // std::ofstream dst(new_path_raw.c_str(), std::ios::binary);
253  // dst << src.rdbuf();
254 
255  // I need to do that way to avoid overwriting if old/new are the same files.
256  std::ifstream source(old_path_raw, std::ios::binary);
257 
258  // file size
259  source.seekg(0, std::ios::end);
260  std::ifstream::pos_type size = source.tellg();
261  source.seekg(0);
262  // allocate memory for buffer
263  char* buffer = new char[size];
264 
265  // copy file
266  source.read(buffer, size);
267  source.close();
268  std::ofstream dest(new_path_raw, std::ios::binary);
269  dest.write(buffer, size);
270 
271  // clean up
272  delete[] buffer;
273  dest.close();
274  }
275 }
