173 void syd::ClinicDatabase::DumpDicom(const std::vector<std::string> & args, std::ostream & os,
174  bool verboseFlag)
175 {
176  // Init the query
177  odb::query<DicomSerie> q = odb::query<DicomSerie>::id.is_not_null();
178 
179  // Build the query with patient (name or study_id)
180  if (args.size() > 0) {
181  std::string n = args[0];
182  if (n != "all") q = q and (odb::query<DicomSerie>::patient->name == n or
183  odb::query<DicomSerie>::patient->study_id == atoi(n.c_str()));
184  }
185 
186  // Build the query with pattern
187  for(auto i=1; i<args.size(); i++) {
188  std::string s = "%"+args[i]+"%";
189  q = q and (odb::query<DicomSerie>::dicom_description.like(s)
190  or odb::query<DicomSerie>::dicom_modality.like(s));
191  }
192 
193  // Sort by patient study_id then acquisition_date
194  q = q + "ORDER BY" + odb::query<DicomSerie>::patient->study_id +
195  "," + odb::query<DicomSerie>::acquisition_date;
196 
197  std::vector<syd::DicomSerie> series;
198  Query<DicomSerie>(q, series);
199  // Dump all information
200  std::vector<syd::Patient> patients;
201  std::vector<std::vector<IdType>> ids;
202  for(auto s:series) {
203  if (patients.size() == 0 or patients.back().id != s.patient->id) {
204  patients.push_back(*s.patient);
205  ids.resize(patients.size());
206  }
207  if (verboseFlag) {
208  os << s.ToString() << std::endl;
209  }
210  ids[patients.size()-1].push_back(s.id);
211  }
212 
213  // Final results
214  if (patients.size() == 0) os << "No dicom match the query." << std::endl;
215  if (patients.size() == 1) os << "One patient match the query: " << std::endl;
216  if (patients.size() > 1) os << patients.size() << " patients match the query: " << std::endl;
217  int i=0;
218  for(auto p:patients) {
219  os << p.study_id << " " << p.name << " with " << ids[i].size() << " series: ";
220  for(auto id:ids[i]) os << id << " ";
221  os << std::endl;
222  i++;
223  }
224 }

299 void syd::ClinicDatabase::DumpSeries(const std::vector<std::string> & query,
300  std::ostream & os)
301 {
302  std::string patientname;
303  if (query.size() == 0) patientname = "";
304  else patientname = query[0];
305  std::vector<Patient> patients;
306  QueryPatientsByName(patientname, patients);
307 
308  PrintTable table;
309  table.AddColumn("#sid");
310  table.AddColumn("name");
311  table.AddColumn("nb",4); // nb of series
312  table.AddColumn("mod",20); // list of modalities (1 CT 3 NM 4 OT)
313  table.Init();
314 
315  for(auto p:patients) {
316  table << p.study_id << p.name;
317  // Get list of series
318  std::vector<DicomSerie> series;
319  Query<DicomSerie>(odb::query<DicomSerie>::patient->id == p.id, series);
320  table << series.size();
321  // Get list of modalities
322  std::map<std::string,int> list;
323  for(auto s:series) {
324  if (list.find(s.dicom_modality) != list.end()) {
325  list[s.dicom_modality]++;
326  }
327  else list[s.dicom_modality] = 1;
328  }
329  std::string mod;
330  for(auto l=list.begin(); l != list.end(); l++) {
331  mod += syd::ToString(l->second)+" "+l->first+" ";
332  }
333  table << mod;
334  }
335  table.Print(os);

607 void syd::ClinicDatabase::CheckSerie_CT(const Serie & serie)
608 {
609  // Check the path exist
610  std::string path = GetPath(serie);
611  if (!syd::DirExists(path)) {
612  LOG(FATAL) << "Error serie " << serie.id << " : the folder " << path << " does not exist.";
613  }
614 
615  // Check file content (slow)
616  if (check_file_content_level_ == 0) return;
617 
618  // Look in the folder
619  OFString scanPattern = "*dcm";
620  OFString dirPrefix = "";
621  OFBool recurse = OFFalse;
622  OFList<OFString> inputFiles;
623  size_t found =
624  OFStandard::searchDirectoryRecursively(path.c_str(), inputFiles, scanPattern, dirPrefix, recurse);
625 
626  // Check the number of files
627  int n = inputFiles.size();
628  if (serie.number_of_files != n) {
629  ELOG(0) << "CheckIntegrity : Error serie " << serie.id << " the number of files is supposed to be " << serie.number_of_files
630  << " but I found " << n << " files int the folder " << path;
631  }
632  if (check_file_content_level_ == 1) return;
633 
634  for(auto i=inputFiles.begin(); i!=inputFiles.end(); i++) {
635  // Open the files
636  DcmFileFormat dfile;
637  syd::OpenDicomFile(i->c_str(), false, dfile);
638  DcmObject *dset = dfile.getDataset();
639 
640  // Check modality
641  std::string modality = GetTagValueString(dset, "Modality");
642  if (serie.modality != modality) {
643  ELOG(0) << "CheckIntegrity : Error serie " << serie.id << " modality is supposed to be " << serie.modality
644  << " but I read " << modality << " in the file " << path << " " << *i;
645  }
646 
647  // Check uid
648  std::string uid = GetTagValueString(dset, "SeriesInstanceUID");
649  if (serie.dicom_uid != uid) {
650  ELOG(0) << "CheckIntegrity : Error serie " << serie.id << " uid is supposed to be " << serie.dicom_uid
651  << " but I read " << uid << " in the file " << path << " " << *i;
652  }
