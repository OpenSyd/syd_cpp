40 syd::Image::pointer
41 syd::StitchDicomImageBuilder::InsertStitchedImage(const syd::DicomSerie::pointer a,
42  const syd::DicomSerie::pointer b)
43 {
44  // Check a and b modality
45  if (a->dicom_modality != b->dicom_modality) {
46  LOG(FATAL) << "Error while trying to stitch the two following images, dicom_modality is not the same: "
47  << std::endl << a
48  << std::endl << b;
49  }
50 
51  // Check a and b patient
52  if (a->patient->id != b->patient->id) {
53  LOG(FATAL) << "Error while trying to stitch the two following images, patient is not the same: "
54  << std::endl << a
55  << std::endl << b;
56  }
57 
58  // Only consider single file and not CT for the moment
59  if (a->dicom_modality == "CT") {
60  LOG(FATAL) << "Error cannot stitch CT images. Only float pixel type (yet).";
61  }
62 
63  // Check if the same frame_of_reference_uid (flag to bypass ?)
64  if (a->dicom_frame_of_reference_uid != b->dicom_frame_of_reference_uid) {
65  LOG(FATAL) << "Error, cannot stitch those two dicoms because the frame_of_reference_uid is not the same:"
66  << std::endl << a << " " << a->dicom_frame_of_reference_uid
67  << std::endl << b << " " << b->dicom_frame_of_reference_uid;
68  }
69 
70  // Create the image record
71  syd::Image::pointer image;
72  db_->New(image);
73  image->patient = a->patient;
74  image->frame_of_reference_uid = a->dicom_frame_of_reference_uid;
75  image->dicoms.push_back(a);
76  image->dicoms.push_back(b);
77  db_->Insert(image); // to obtain an id
78 
79 
80  std::ostringstream oss;
81  oss << a->dicom_modality << "_" << image->id << ".mhd";
82  std::string mhd_filename = oss.str();
83  std::string mhd_relative_path = image->ComputeRelativeFolder()+PATH_SEPARATOR;
84  std::string mhd_path = db_->ConvertToAbsolutePath(mhd_relative_path+mhd_filename);
85  image->UpdateFile(db_, mhd_filename, mhd_relative_path);
86 
87  // Auto tag
88  syd::Tag::pointer tag_stitch = db_->FindOrInsertTag("stitch", "Image computed by stitching 2 images");
89  image->AddTag(tag_stitch);
90 
91  // Read the dicom images
92  typedef float PixelType;
93  typedef itk::Image<PixelType, 3> ImageType;
94  ImageType::Pointer image_a = db_->ReadImage<PixelType>(a);
95  ImageType::Pointer image_b = db_->ReadImage<PixelType>(b);
96 
97  // Stitch (default values for now, to be changed!)
98  ImageType::Pointer output = syd::StitchImages<ImageType>(image_a, image_b, 150000, 4);
99 
100  // Update the image values
101  db_->UpdateImageInfo<PixelType>(image, output, true); // true = update md5
102 
103  // Write image
104  syd::WriteImage<ImageType>(output, db_->GetAbsolutePath(image));
105 
106  // Insert into the db
107  db_->Update(image);
108  return image;
109 }
