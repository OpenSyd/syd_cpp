42 syd::Image syd::ImageBuilder::InsertImageFromDicomSerie(const syd::DicomSerie & dicomserie)
43 {
44  if (tag_.label == "unamed_tag") {
45  EXCEPTION("Error in ImageBuilder, use SetImageTag before");
46  }
47 
48  // Get the files
49  std::vector<syd::DicomFile> dicom_files;
50  db_->Query<syd::DicomFile>(odb::query<syd::DicomFile>::dicom_serie->id == dicomserie.id, dicom_files);
51  if (dicom_files.size() == 0) {
52  EXCEPTION("Error not DicomFile associated with this DicomSerie: " << dicomserie);
53  }
54  std::vector<std::string> dicom_filenames;
55  for(auto f:dicom_files) {
56  dicom_filenames.push_back(db_->GetAbsolutePath(*f.file));
57  }
58 
59  // Get patient
60  syd::Patient patient = *dicomserie.patient;
61 
62  // Create image first to get the id
63  syd::Image image;
64  image.patient = std::make_shared<syd::Patient>(patient);
65  image.tag = std::make_shared<syd::Tag>(tag_);
66  image.dicoms.push_back(std::make_shared<syd::DicomSerie>(dicomserie));
67  image.type = "mhd";
68  db_->Insert(image);
69 
70  // Define filename, path
71  std::string path = db_->GetAbsoluteFolder(patient);
72  std::ostringstream oss;
73  oss << dicomserie.dicom_modality << "_" << image.id << ".mhd";
74  std::string filename = oss.str();
75  std::string output = path+PATH_SEPARATOR+filename;
76 
77  // Convert dicom, write to disk
78  std::string md5;
79  try {
80  if (dicom_filenames.size() == 1) { // probably OT, NM image
81  typedef float PixelType;
82  typedef itk::Image<PixelType, 3> ImageType;
83  ImageType::Pointer itk_image = syd::ReadDicomFromSingleFile(dicom_filenames[0]);
84  syd::WriteImage<ImageType>(itk_image, output);
85  UpdateImageInfo<ImageType>(image, itk_image);
86  md5 = ComputeImageMD5<ImageType>(itk_image);
87  }
88  if (dicomserie.dicom_modality == "CT") {
89  typedef short PixelType;
90  typedef itk::Image<PixelType, 3> ImageType;
91  std::string folder = dicom_filenames[0];
92  syd::Replace(folder, GetFilenameFromPath(folder), "");
93  ImageType::Pointer itk_image = syd::ReadDicomSerieFromFolder(folder, dicomserie.dicom_series_uid);
94  syd::WriteImage<ImageType>(itk_image, output);
95  UpdateImageInfo<ImageType>(image, itk_image);
96  md5 = ComputeImageMD5<ImageType>(itk_image);
97  }
98 
99  // Create File and Image object
100  syd::File mhd;
101  mhd.filename = filename;
102  mhd.path = db_->GetRelativeFolder(patient);
103  syd::Replace(filename, ".mhd", ".raw");
104  syd::File raw;
105  raw.filename = filename;
106  raw.path = db_->GetRelativeFolder(patient);
107  raw.md5 = md5;
108  db_->Insert(mhd);
109  db_->Insert(raw);
110  image.files.push_back(std::make_shared<syd::File>(mhd));
111  image.files.push_back(std::make_shared<syd::File>(raw)); // must be after file insertion
112  db_->Update(image);
113  } catch (syd::Exception & e) {
114  db_->Delete<syd::Image>(image);
115  EXCEPTION("Error during CreateImageFromDicomSerie: " << e.what() << " (I remove the image " << image << ")");
116  }
117 
118  return image;
119 }
