41 syd::Image syd::ImageBuilder::InsertImage(const syd::DicomSerie & dicomserie)
42 {
43  // Get the files
44  LOG(4) << "Get the DicomFiles from " << dicomserie;
45  std::vector<syd::DicomFile> dicom_files;
46  db_->Query<syd::DicomFile>(odb::query<syd::DicomFile>::dicom_serie->id == dicomserie.id, dicom_files);
47  if (dicom_files.size() == 0) {
48  EXCEPTION("Error not DicomFile associated with this DicomSerie: " << dicomserie);
49  }
50  std::vector<std::string> dicom_filenames;
51  for(auto f:dicom_files) {
52  dicom_filenames.push_back(db_->GetAbsolutePath(f));
53  }
54  LOG(4) << "Found " << dicom_files.size();
55 
56  // Create image first to get the id
57  syd::Image image = InsertEmptyImage(*dicomserie.patient);
58  LOG(4) << "Update image dicom";
59  UpdateDicom(image, dicomserie);
60  LOG(4) << "Update image files";
61  // UpdateFile(image, GetDefaultImageRelativePath(image), GetDefaultImageFilename(image));
62  UpdateFile(image, db_->ComputeRelativeFolder(image), GetDefaultImageFilename(image));
63  std::string f = db_->GetAbsolutePath(image);
64 
65  // Convert dicom, write to disk
66  // std::string md5;
67  try {
68  if (dicomserie.dicom_modality == "CT") {
69  typedef short PixelType;
70  typedef itk::Image<PixelType,3> ImageType;
71  LOG(4) << "Read dicom";
72  ImageType::Pointer itk_image = ReadImage<PixelType>(dicomserie);
73  LOG(4) << "Update image info and md5";
74  UpdateImageInfo<PixelType>(image, itk_image, true); // true = update md5
75  LOG(4) << "Write image on disk";
76  syd::WriteImage<ImageType>(itk_image, f);
77  // Auto tag the image with ct
78  syd::Tag tag = db_->FindOrInsertTag("ct", "CT image");
79  image.AddTag(tag);
80  }
81  else {
82  // No auto tag, we dont know if this is a spect, a pet etc
83  typedef float PixelType;
84  typedef itk::Image<PixelType,3> ImageType;
85  ImageType::Pointer itk_image = ReadImage<PixelType>(dicomserie);
86  UpdateImageInfo<PixelType>(image, itk_image, true); // true = update md5
87  syd::WriteImage<ImageType>(itk_image, f);
88  }
89 
90  // Insert in the db
91  LOG(4) << "Update database.";
92  db_->Update(image);
93 
94  } catch (syd::Exception & e) {
95  db_->Delete(image);
96  EXCEPTION("Error during InsertImage: " << e.what() << " (I remove the image " << image << ")");
97  }
98 
99  return image;
100 }

105 syd::Image syd::ImageBuilder::InsertStitchedImage(const syd::DicomSerie & a,
106  const syd::DicomSerie & b)
107 {
108  // Check a and b modality
109  if (a.dicom_modality != b.dicom_modality) {
110  LOG(FATAL) << "Error while trying to stitch the two following images, dicom_modality is not the same: "
111  << std::endl << a
112  << std::endl << b;
113  }
114 
115  // Only consider single file and not CT for the moment
116  if (a.dicom_modality == "CT") {
117  LOG(FATAL) << "Error cannot stitch CT images. Only float pixel type (yet).";
118  }
119 
120  // Check if the same frame_of_reference_uid (flag to bypass ?)
121  if (a.dicom_frame_of_reference_uid != b.dicom_frame_of_reference_uid) {
122  LOG(FATAL) << "Error, cannot stitch those two dicoms because the frame_of_reference_uid is not the same:"
123  << std::endl << a << " " << a.dicom_frame_of_reference_uid
124  << std::endl << b << " " << b.dicom_frame_of_reference_uid;
125  }
126 
127  // Create the image record
128  syd::Image image = InsertEmptyImage(*a.patient);
129  UpdateDicom(image, a);
130  UpdateDicom(image, b);
131  // UpdateFile(image, GetDefaultImageRelativePath(image), GetDefaultImageFilename(image));
132  UpdateFile(image, db_->ComputeRelativeFolder(image), GetDefaultImageFilename(image));
133 
134  // Auto tag
135  syd::Tag tag_stitch = db_->FindOrInsertTag("stitch", "Image computed by stitching 2 images");
136  image.AddTag(tag_stitch);
137 
138  // Read the dicom images
139  typedef float PixelType;
140  typedef itk::Image<PixelType, 3> ImageType;
141  ImageType::Pointer image_a = ReadImage<PixelType>(a);
142  ImageType::Pointer image_b = ReadImage<PixelType>(b);
143 
144  // Stitch (default values for now, to be changed!)
145  ImageType::Pointer output = syd::StitchImages<ImageType>(image_a, image_b, 150000, 4);
146 
147  // Update the image values
148  UpdateImageInfo<PixelType>(image, output, true); // true = update md5
149 
150  // Write image
151  syd::WriteImage<ImageType>(output, db_->GetAbsolutePath(image));
152 
153  // Insert into the db
154  db_->Update(image);
155  return image;
156 }

288 syd::RoiMaskImage syd::ImageBuilder::InsertRoiMaskImage(const syd::DicomSerie & dicom,
289  const syd::RoiType & roitype,
290  const std::string & filename)
291 {
292  // Create a mask
293  syd::RoiMaskImage mask;
294  mask.roitype = std::make_shared<syd::RoiType>(roitype);
295  std::shared_ptr<syd::Image> image;
296 
297  try {
298  // Create and insert the associated image (must be a pointer here because will bne updated)
299  image = std::make_shared<syd::Image>(InsertEmptyImage(*dicom.patient));
300  mask.image = image;
301 
302  // Add auto tag (mask)
303  syd::Tag tag_mask = db_->FindOrInsertTag("mask", "Mask image");
304  image->AddTag(tag_mask);
305 
306  // Need to insert the mask to get an id
307  db_->Insert(mask);
308 
309  // Update info
310  UpdateDicom(*image, dicom);
311 
312  // Need to set the image of the mask and insert it before GetDefaultRoiMaskImageRelativePath (need image patient)
313  // UpdateFile(*image, GetDefaultRoiMaskImageRelativePath(mask), GetDefaultRoiMaskImageFilename(mask));
314  UpdateFile(*image, db_->ComputeRelativeFolder(mask), GetDefaultRoiMaskImageFilename(mask));
315 
316  // Create image folder if needed
317  std::string relative_folder = db_->ComputeRelativeFolder(*image);
318  std::string absolute_folder = db_->ConvertToAbsolutePath(relative_folder);
319  if (!syd::DirExists(absolute_folder)) syd::CreateDirectory(absolute_folder);
320 
321  // Read image
322  typedef unsigned char PixelType;
323  typedef itk::Image<PixelType,3> ImageType;
324  ImageType::Pointer itk_image;
325  itk_image = syd::ReadImage<ImageType>(filename);
326 
327  // Update info
328  UpdateImageInfo<PixelType>(*image, itk_image, true); // true = update md5
329 
330  // Write image
331  syd::WriteImage<ImageType>(itk_image, db_->GetAbsolutePath(*image));
332 
333  }
334  catch(std::exception & e) {
335  // delete (silently)
336  int l = Log::LogLevel();
337  Log::LogLevel() = 0;
338  db_->Delete(mask);
339  Log::LogLevel() = l;
340  EXCEPTION("Error in 'InsertRoiMaskImage' with dicom=" << dicom.id
341  << " roitype='" << roitype.name << "' file=[" << filename << "]"
342  << std::endl << e.what());
343  }
344 
345  // Insert into the db
346  db_->Update(*image);
347  db_->Update(mask);
348  return mask;
349 }

354 void syd::ImageBuilder::CropImageLike(syd::Image & image, syd::Image & like, bool forceFlag)
355 {
356  // Check
357  if (!forceFlag) {
358  std::string ref="";
359  for(auto d:like.dicoms) {
360  if (ref != "") {
361  if (ref != d->dicom_frame_of_reference_uid) {
362  LOG(WARNING) << "Image associated with several dicom_frame_of_reference_uid. " << like;
363  }
364  }
365  ref = d->dicom_frame_of_reference_uid;
366  }
367  std::string ref2="";
368  for(auto d:image.dicoms) {
369  if (ref2 != "") {
370  if (ref2 != d->dicom_frame_of_reference_uid) {
371  LOG(WARNING) << "Image associated with several dicom_frame_of_reference_uid. " << like;
372  }
373  }
374  ref2 = d->dicom_frame_of_reference_uid;
375  }
376  if (ref != ref2) {
377  LOG(FATAL) << "Cannot crop: " << image << std::endl << "like: "
378  << like << std::endl
379  << "because they dont share the same dicom_frame_of_reference_uid (coordinate system).";
380  }
381  }
382 
383  // Crop the image and write it to disk
384  if (image.pixel_type == "float") CropImageLike<float>(image, like);
385  else if (image.pixel_type == "short") CropImageLike<short>(image, like);
386  else if (image.pixel_type == "unsigned char") CropImageLike<unsigned char>(image, like);
387  else {
388  LOG(FATAL) << "Unknown pixel_type: " << image.pixel_type;
389  }
390 }
