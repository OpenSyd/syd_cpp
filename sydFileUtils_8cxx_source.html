57 bool syd::EqualFiles(std::ifstream & in1, std::ifstream & in2)
58 {
59  std::ifstream::pos_type size1, size2;
60  size1 = in1.seekg(0, std::ifstream::end).tellg();
61  in1.seekg(0, std::ifstream::beg);
62  size2 = in2.seekg(0, std::ifstream::end).tellg();
63  in2.seekg(0, std::ifstream::beg);
64  if(size1 != size2) return false;
65 
66  static const size_t BLOCKSIZE = 4096;
67  size_t remaining = size1;
68  while(remaining) {
69  char buffer1[BLOCKSIZE], buffer2[BLOCKSIZE];
70  size_t size = std::min(BLOCKSIZE, remaining);
71  in1.read(buffer1, size);
72  in2.read(buffer2, size);
73  if(0 != memcmp(buffer1, buffer2, size)) return false;
74  remaining -= size;
75  }
76  return true;
77 }

200 int syd::GetPageContent(char const *argv[], std::ostream & os) {
201  // http://stackoverflow.com/questions/27491288/read-http-file-content-to-string-with-boost-c
202  LOG(1) << "Try to connect to " << argv[0] << argv[1];
203  try {
204  boost::asio::io_service io_service;
205  // Get a list of endpoints corresponding to the server name.
206  tcp::resolver resolver(io_service);
207  tcp::resolver::query query(argv[0], "http");
208  tcp::resolver::iterator endpoint_iterator = resolver.resolve(query);
209  tcp::resolver::iterator end;
210  // Try each endpoint until we successfully establish a connection.
211  tcp::socket socket(io_service);
212  boost::system::error_code error = boost::asio::error::host_not_found;
213  while (error && endpoint_iterator != end) {
214  socket.close();
215  socket.connect(*endpoint_iterator++, error);
216  }
217  if (error)
218  throw boost::system::system_error(error);
219  // Form the request. We specify the "Connection: close" header so that the
220  // server will close the socket after transmitting the response. This will
221  // allow us to treat all data up until the EOF as the content.
222  boost::asio::streambuf request;
223  std::ostream request_stream(&request);
224  request_stream << "GET " << argv[1] << " HTTP/1.0\r\n";
225  request_stream << "Host: " << argv[0] << "\r\n";
226  request_stream << "Accept: */*\r\n";
227  request_stream << "Connection: close\r\n\r\n";
228  // Send the request.
229  boost::asio::write(socket, request);
230  // Read the response status line.
231  boost::asio::streambuf response;
232  boost::asio::read_until(socket, response, "\r\n");
233  // Check that response is OK.
234  std::istream response_stream(&response);
235  std::string http_version;
236  response_stream >> http_version;
237  unsigned int status_code;
238  response_stream >> status_code;
239  std::string status_message;
240  std::getline(response_stream, status_message);
241  if (!response_stream || http_version.substr(0, 5) != "HTTP/") {
242  std::cerr << "Invalid response\n";
243  return 1;
244  }
245  if (status_code != 200) {
246  std::cerr << "Response returned with status code " << status_code << "\n";
247  return 1;
248  }
249 
250  // Read the response headers, which are terminated by a blank line.
251  boost::asio::read_until(socket, response, "\r\n\r\n");
252 
253  // Process the response headers.
254  std::string header;
255 
256  while (std::getline(response_stream, header) && header != "\r")
257  LOG(2) << header;
258 
259  LOG(2) << "Writing content data";
260  // Write whatever content we already have to output.
261  if (response.size() > 0) {
262  os << &response;
263  }
264 
265  // Read until EOF, writing data to output as we go.
266  while (true) {
267  size_t n = boost::asio::read(socket, response, boost::asio::transfer_at_least(1), error);
268 
269  if (!error)
270  {
271  if (n)
272  os << &response;
273  }
274 
275  if (error == boost::asio::error::eof)
276  break;
277 
278  if (error)
279  throw boost::system::system_error(error);
280  }
281  }
282  catch (std::exception &e) {
283  std::cerr << "Exception, could not connect to " << argv[0] << " exception is : " << e.what() << "\n";
284  }
285 
286  return 0;
287 }
